8) Graphs

BFS / DFS / topo / DSU

Adj list:

g = defaultdict(list)
for u, v in edges:
    g[u].append(v)
    g[v].append(u)   # if undirected

BFS (shortest in unweighted graph):

def bfs(start):
    q = deque([start])
    dist = {start: 0}
    while q:
        u = q.popleft()
        for v in g[u]:
            if v not in dist:
                dist[v] = dist[u] + 1
                q.append(v)
    return dist

DFS (iterative to avoid recursion limit):

stack = [start]
seen = set([start])
while stack:
    u = stack.pop()
    for v in g[u]:
        if v not in seen:
            seen.add(v)
            stack.append(v)

Topological sort (Kahn):

indeg = Counter()
for u in g:
    for v in g[u]:
        indeg[v] += 1

q = deque([u for u in nodes if indeg[u] == 0])
order = []
while q:
    u = q.popleft()
    order.append(u)
    for v in g[u]:
        indeg[v] -= 1
        if indeg[v] == 0:
            q.append(v)

Union-Find (DSU):

class DSU:
    def __init__(self, n):
        self.p = list(range(n))
        self.r = [0]*n
    def find(self, x):
        while self.p[x] != x:
            self.p[x] = self.p[self.p[x]]
            x = self.p[x]
        return x
    def union(self, a, b):
        ra, rb = self.find(a), self.find(b)
        if ra == rb: return False
        if self.r[ra] < self.r[rb]:
            ra, rb = rb, ra
        self.p[rb] = ra
        if self.r[ra] == self.r[rb]:
            self.r[ra] += 1
        return True

    if len(path) == n:
        res.append(path[:]); return
    for i in range(n):

        used[i] = True
        path.append(a[i])
        bt()
        path.pop()
